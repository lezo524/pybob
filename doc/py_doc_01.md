word = 'Python'
word[-1] --> 마지막 문자
word[:2] --> character from the beginning to position 2 (excluded)
'Py'
word[4:] --> characters from position 4 (included) to the end
'on'
word[-2] --> characters from the second-last (included) to the end
'on'
word[:2] + word[2:]
'Python'
len(word)
6
